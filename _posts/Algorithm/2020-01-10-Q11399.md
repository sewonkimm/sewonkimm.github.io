#include <stdio.h>

void bubble_sort(int arr[], int n) {
	int temp;

	for (int i = 0; i<n - 1; i++) {
		for (int j = n - 1; j>i; j--) {
			if (arr[j - 1] > arr[j]) {
				temp = arr[j];
				arr[j] = arr[j - 1];
				arr[j - 1] = temp;
			}
		}
	}
}

int main() {

	int t[1000] = {};

	int p;
	scanf("%d", &p);

	for (int i=0; i<p; i++) {
		scanf("%d ", &t[i]);
	}

	bubble_sort(t, p);

	int sum = 0;
	int sum2 = 0;
	for (int i = 0; i<p; i++) {
		sum += t[i];
		sum2 += sum;
	}

	printf("%d", sum2);


	return 0;
}
