## Code

#include <iostream>
#include <queue>
#include <string.h>
#include <vector>
using namespace std;

vector<int> from(100001);

// ì—­ìˆœ ì¶œë ¥ì„ ìœ„í•œ í•¨ìˆ˜
void print(int n, int m) {
	if (n != m) {
		print(n, from[m]);
	}

	cout << m << " ";
}

int main() {
	
	int n, k;
	cin >> n >> k;

	int check[100001];
	int dist[100001];
	memset(check, 0, sizeof(check));
	memset(dist, 0, sizeof(dist));

	queue<int> q;
	q.push(n);
	while (!q.empty()) {
		int now = q.front();
		check[now] = 1;
		q.pop();

		if (now - 1 >= 0) {
			if (check[now - 1] == 0) {
				q.push(now - 1);
				check[now - 1] = 1;
				from[now - 1] = now;
				dist[now - 1] = dist[now] + 1;
			}
		}
		
		if(now +1 <= 100000) {
			if (check[now + 1] == 0) {
				q.push(now + 1);
				check[now + 1] = 1;
				from[now + 1] = now;
				dist[now + 1] = dist[now] + 1;
			}
		}
		
		if (2 * now <= 100000) {
			if (check[2 * now] == 0) {
				q.push(2 * now);
				check[2 * now] = 1;
				from[2 * now] = now;
				dist[2 * now] = dist[now] + 1;
			}
		}

	}
	
	cout << dist[k] << "\n";
	
	print(n, k);
	
	return 0;
}
