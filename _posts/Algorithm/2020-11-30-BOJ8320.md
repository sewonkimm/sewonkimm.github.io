    public class Main {

        public static void main(String[] args) {
            Scanner sc = new Scanner(System.in);
            int n = sc.nextInt();
            int count = 0;
            
            for (int i = 1; i <= n; i++) {
                for (int x = i; i*x <= n ; x++) {
                    count++;
                }
            }
