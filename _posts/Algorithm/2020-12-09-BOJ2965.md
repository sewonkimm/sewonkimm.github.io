import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int a = sc.nextInt();
		int b = sc.nextInt();
		int c = sc.nextInt();

		boolean[] map = new boolean[100];
		map[a] = true;
		map[b] = true;
		map[c] = true;

		jump(map, 0);
		sc.close();
	}

	private static void jump(boolean[] map, int count) {

		int[] num = new int[3];
		int idx = 0;
		for (int i = 0; i < map.length; i++) {
			if(idx == 3)	break;

			if(map[i])	num[idx++] = i;
		}

		// ë” ì´ìƒ ì´ë™í•  ìˆ˜ ì—†ëŠ” ê²½ìš°

		}
		else {
			map[num[0]] = false;
			map[num[1]+1] = true;
		}
		jump(map, count+1);
	}

}
