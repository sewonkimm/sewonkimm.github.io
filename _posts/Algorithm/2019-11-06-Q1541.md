## Code
```cpp
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main(int argc, char **argv)
{
	string str;
	cin >> str;

	string temp;
	bool minus = false;
	int answer = 0;
	for (int i = 0; i <= str.size(); i++) {

		// +, - ì¼ ë•Œ
		if (str[i] == '+' || str[i] == '-' || str[i] == '\0') {
			if (minus) {
				answer -= stoi(temp);
			}
			else {
				answer += stoi(temp);
			}
			temp = "";
		}

		
		if (str[i] == '-') {
			minus = true;
			continue;
		}
		
		
		// ìˆ«ì ì¼ ë•Œ
		temp += str[i];
	}

	cout << answer << endl;

	return 0;
}
```
