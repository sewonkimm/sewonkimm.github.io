vector<int> solution(vector<int> heights) {
    vector<int> answer(heights.size());

    // ë’¤ì—ì„œ ë¶€í„° ì•ìœ¼ë¡œ ê²€ì‚¬

        int now = heights[i];

        int receive = 0;
        for(int j=i-1; j>=0; j--){
            if(heights[j] > now) {
                receive = j+1;
                break;
            }
        }

        answer[i] = receive;
    }
    return answer;
}
