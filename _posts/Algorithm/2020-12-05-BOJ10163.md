import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int[][] map = new int[101][101];
		int[] count = new int[N+1];

		for (int i = 1; i <= N; i++) {
			int x = sc.nextInt();
			int y = sc.nextInt();
			int w = sc.nextInt();
			int h = sc.nextInt();

			for (int col = 0; col < w; col++) {
				for (int row = 0; row < h; row++) {
					map[x+col][y+row] = i;
				}
			}
		}

		for (int x = 0; x <= 100; x++) {
			for (int y = 0; y <= 100; y++) {
				int num = map[x][y];
				if(num > 0)	count[num]++;
			}
		}

		for(int i = 1; i <= N; i++) {
			System.out.println(count[i]);
		}
		sc.close();
	}

}
