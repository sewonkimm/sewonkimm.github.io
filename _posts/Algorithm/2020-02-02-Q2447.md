#include <iostream>

using namespace std;

static char map[10000][10000];

void isStar(int x, int y, int n){
	int m = n/3;
	
	if(n == 1){
		map[x][y] = '*';
	}
	else {
		for(int i=0; i<3; i++){
			for(int j=0; j<3; j++){

		}
	}
}
