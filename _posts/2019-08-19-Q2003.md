## Code

```cpp
#include<iostream>
using namespace std;

int a[10001];

int main() {

	int n;
	long long int m;
	cin >> n >> m;

	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}

	int left, right;
	left = right = 0;

	long long int sum = a[0];
	int cnt = 0;

	while (left <= right && right < n) {
		if(sum < m){
			right += 1;
			sum += a[right];
		}
		else if (sum > m) {

			if (left != right) {
				sum -= a[left];
				left += 1;
			}
			else {
				right += 1;
				sum += a[right];
			}

		}
		else if (sum == m) {
			cnt += 1;
			right += 1;
			sum += a[right];
		}

	}

	cout << cnt << "\n";



	return 0;
}
```
