## Code
```cpp
#include <iostream>
#include <queue>
using namespace std;

int check[100000];
int dist[100000];

int main() {

	int n, k;
	cin >> n >> k;
	
	queue<int> q;
	q.push(n);
	while (!q.empty()) {
		int now = q.front();
		check[now] = true;
		q.pop();

		int next1 = now - 1;
		if (next1 >= 0) {
			if (check[next1] == 0) {
				check[next1] = true;
				q.push(next1);
				dist[next1] = dist[now] + 1;
			}
		}

		int next2 = now + 1;
		if (next2 < 100001) {
			if (check[next2] == 0) {
				check[next2] = true;
				q.push(next2);
				dist[next2] = dist[now] + 1;
			}
		}

		int next3 = 2 * now;
		if (next3 < 100001) {
			if (check[next3] == 0) {
				check[next3] = true;
				q.push(next3);
				dist[next3] = dist[now] + 1;
			}
		}
		
	}
		
	cout << dist[k] << "\n";
	
	return 0;
}
```
