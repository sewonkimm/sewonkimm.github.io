import java.util.Scanner;

public class BOJ2098 {

	static int N;
	static int[][] W;
	static int[] visited;
	static int min;
	public static void main(String[] args) {
		// input
		Scanner sc = new Scanner(System.in);
		N = sc.nextInt();	// ë„ì‹œ ê°¯ìˆ˜
		W = new int[N][N];	// ë¹„ìš©
		for(int i=0; i<N; i++) {
			for(int j=0; j<N; j++) {
				W[i][j] = sc.nextInt();
			}
		}
		min = Integer.MAX_VALUE;

		for(int i=0; i<N; i++) {
			visited = new int[N];
			visited[i] = 1;
			tsp(i, 0, 0, 1);
		}

		// output
		System.out.println(min);
		sc.close();
	}


	private static void tsp(int start, int location, int sum, int cnt) {
		if(cnt == N) {	// nê°œì˜ ë„ì‹œë¥¼ ì „ë¶€ ë°©ë¬¸í•œ ê²½ìš°

			if(W[location][start] == 0)	return;	// ë‹¤ì‹œ ì‹œì‘ë„ì‹œë¡œ ê°€ëŠ” ê¸¸ì´ ì—†ìŒ

			sum +=	W[location][start];	// ë‹¤ì‹œ ì‹œì‘ë„ì‹œë¡œ ê°
			min = Math.min(min, sum);
			return;
		}

		for(int i=0; i<N; i++) {
			if(visited[i] == 0 && W[location][i] != 0) {
				visited[i] = 1;
				tsp(start, i, sum+W[location][i] , cnt+1);
				visited[i] = 0;
			}
		}
	}
}
