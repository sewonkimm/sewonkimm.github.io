int main() {

	int n;
	cin >> n;
	
	int d[100001];
	memset(d, 0, sizeof(d));

	for (int i = 1; i <= n; i++) {
		d[i] = i;

		for (int j = 1; j*j <= i; j++) {
			if (d[i] > d[i - (j*j)] + 1) {

		}

	}

	cout << d[n] << "\n";
	
	return 0;
}
