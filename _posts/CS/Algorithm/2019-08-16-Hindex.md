int solution(vector<int> citations) {
    int answer = 0;

    sort(citations.begin(), citations.end());
    reverse(citations.begin(), citations.end());

    for(int i=0; i<=citations.size(); i++){
        if(i >= citations[i]) {
            answer = i;
            break;

            return i;
        }
    }
