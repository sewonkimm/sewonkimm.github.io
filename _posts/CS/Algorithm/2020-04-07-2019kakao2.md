#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int findNum(vector<int> answer, vector<int> nums){
    for(int i=0; i<answer.size(); i++){
        int x = answer[i];
        
        // find X & swap
        for(int j=0; j<nums.size(); j++){
            if(nums[j] == x){
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
            }
        }
    }
    return nums[nums.size()-1];
}

vector<int> solution(string s) {
    
    vector< vector<int> > nums;
    vector<int> temp;

            temp.push_back(stoi(num));
            nums.push_back(temp);
            temp.clear();
            isNum = false;
        }
        else if(s[i] == ',' && isNum){
            temp.push_back(stoi(num));
            num = "";
        }

            num += s[i];
        }

            answer.push_back(nums[target][0]);
        }
        else {
