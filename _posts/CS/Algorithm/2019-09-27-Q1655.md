			max_heap.push(num);
		}
		else {
			min_heap.push(num);
		}

		// swap
		if (!min_heap.empty() && max_heap.top() > min_heap.top()) {
			int maxtop = max_heap.top();
			int mintop = min_heap.top();

			max_heap.pop();
			min_heap.pop();

			max_heap.push(mintop);
			min_heap.push(maxtop);
		}

		printf("%d\n", max_heap.top());
	}

	return 0;
}
