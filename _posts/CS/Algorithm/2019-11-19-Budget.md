## Code
```cpp
#include <iostream>
#include <stdio.h>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<int> d, int budget) {
    sort(d.begin(), d.end());
    
    int answer = 0;
    int i = 0;
    while(budget > 0){
        if(budget - d[i] >= 0){
            answer++;
            budget -= d[i];
        }
        i++;
        if(i == d.size()){
            break;
        }
    }
    return answer;
}
```
