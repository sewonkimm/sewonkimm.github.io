## Code

```cpp
#include<iostream>
#include<queue>
using namespace std;

bool check[200001];
int dist[200001];
int cnt[200001];

int main() {

	int n, k;
	cin >> n >> k;

	queue<int> q;
	q.push(n);
	check[n] = true;
	cnt[n] = 1;

	while (!q.empty()) {
		int now = q.front();
		q.pop();

		for (int next : { now - 1, now + 1, now * 2 }) {
			if (next >= 0 && next <= 200000) {
				//λ‹¤μ λ…Έλ“κ°€ λ°©λ¬Έν•μ§€ μ•μ€ λ…Έλ“μΌ κ²½μ°
				if (check[next] == false) {
					q.push(next);
					check[next] = true;
					dist[next] = dist[now] + 1;
					cnt[next] = cnt[now];

				}
				//λ‹¤μ λ…Έλ“λ¥Ό μ΄λ―Έ λ°©λ¬Έν–μ„ κ²½μ°
				else if (dist[next] == dist[now] + 1) {
					cnt[next] += cnt[now];
				}
			}
		}
	}

	cout << dist[k] << "\n" << cnt[k] << "\n";
	return 0;
}
```
