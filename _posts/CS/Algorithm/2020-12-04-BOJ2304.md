import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int[] stick = new int[1001];
		for (int i = 0; i < N; i++) {
			int x = sc.nextInt();
			stick[x] = sc.nextInt();
		}
		
		int x = getHigest(stick);
		
		int sum = 0;
		// ì¢Œì¸¡
		int max = stick[0];
		for (int i = 0; i < x; i++) {
			if(stick[i] > max) {
				max = stick[i];
			}
			sum += max;
		}
		
		// ìš°ì¸¡
		max = stick[1000];
		for(int i=1000; i > x; i--) {
			if(stick[i] > max) {
				max = stick[i];
			}
			sum += max;
		}
		
		sum += stick[x];
		System.out.println(sum);
		sc.close();
	}

	private static int getHigest(int[] stick) {
		int max = Integer.MIN_VALUE;
		int maxIndex = 0;
		for (int i = 0; i <= 1000; i++) {
			if(stick[i] > max) {
				max = stick[i];
				maxIndex = i;
			}
		}
		return maxIndex;
	}

}
