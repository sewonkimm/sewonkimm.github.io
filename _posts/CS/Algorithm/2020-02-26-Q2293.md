#include<vector>
#include<algorithm>

using namespace std;

int main(){
	int n, k;
	cin >> n >> k;
	
	vector<int> c(n+1, 0);	
	for(int i=1; i<=n; i++){
		cin >> c[i];
	}
	sort(c.begin(), c.end());
	
	vector< vector<int> > d(n+1, vector<int>(k+1, 0));
	for(int i=0; i<=n; i++){
		d[i][0] = 1;
	}
	
	for(int i=1; i<=n; i++){
		for(int j=1; j<=k; j++){

			if(j >= c[i]){
				d[i][j] = d[i-1][j] + d[i][j-c[i]];
			}
			else {
				d[i][j] = d[i-1][j];
			}
		}
	}
	
	cout << d[n][k] << endl;
	
	return 0;
}
