#include<iostream>
#include<vector>
using namespace std;

bool isPrime(int x) {
	if (x < 2) {
		return false;
	}

	for (int i = 2; i*i <= x; i++) {
		if (x % i == 0) {
			return false;
		}
	}

	return true;
}
int main() {

	int n;
	cin >> n;

	//ì†Œìˆ˜ êµ¬í•˜ê¸°
	vector<int> prime;
	for (int i = 2; i <= n; i++) {
		if (isPrime(i)) {
			prime.push_back(i);
		}
	}
	prime.push_back(0);

	//ì—°ì†í•© êµ¬í•˜ê¸°
	int cnt = 0;
	int left, right;
	left = right = 0;
	int sum = prime[0];
	while (left <= right && right < prime.size()-1) {
		if (sum < n) {
			right += 1;
			sum += prime[right];
		}
		else if (sum > n) {
			if (left == right) {
				right += 1;
				sum += prime[right];
			}
			else {
				sum -= prime[left];
				left += 1;
			}
		}
		else if (sum == n) {
			cnt++;
			right += 1;
			sum += prime[right];
		}
	}

	cout << cnt << "\n";

	return 0;
}
